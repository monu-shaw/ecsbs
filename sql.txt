CREATE TABLE `ecsbs`.`seller` (`id` INT NOT NULL AUTO_INCREMENT , `sellerName` VARCHAR(255) NOT NULL , `businessName` VARCHAR(100) NOT NULL , `businessAddress` VARCHAR(225) NOT NULL , `country` VARCHAR(50) NOT NULL , `currency` VARCHAR(50) NOT NULL , `email` VARCHAR(100) NOT NULL , `phone` VARCHAR(50) NOT NULL , `password` VARCHAR(255) NOT NULL , `slug` VARCHAR(100) NOT NULL , `createdOn` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
