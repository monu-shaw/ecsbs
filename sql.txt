CREATE TABLE `ecsbs`.`seller` (`id` INT NOT NULL AUTO_INCREMENT , `sellerName` VARCHAR(255) NOT NULL , `businessName` VARCHAR(100) NOT NULL , `businessAddress` VARCHAR(225) NOT NULL , `country` VARCHAR(50) NOT NULL , `currency` VARCHAR(50) NOT NULL , `email` VARCHAR(100) NOT NULL , `phone` VARCHAR(50) NOT NULL , `password` VARCHAR(255) NOT NULL , `slug` VARCHAR(100) NOT NULL , `createdOn` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;

CREATE TABLE `ecsbs`.`product` (`id` INT(25) NOT NULL AUTO_INCREMENT , `name` VARCHAR(255) NOT NULL , `MeasuringSize` VARCHAR(255) NOT NULL , `MeasuringUnit` VARCHAR(255) NOT NULL , `Image` VARCHAR(500) NOT NULL , `Description` VARCHAR(1000) NOT NULL , `CategoryId` INT NOT NULL , `SellerId` INT NOT NULL , `slug` VARCHAR(250) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

CREATE TABLE `ecsbs`.`category` (`id` INT NOT NULL AUTO_INCREMENT , `name` VARCHAR(100) NOT NULL , `Image` VARCHAR(255) NOT NULL , `sellerId` INT NOT NULL , `slug` VARCHAR(100) NOT NULL , `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
